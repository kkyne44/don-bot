CREATE TABLE IF NOT EXISTS DON_USER(
    UserId VARCHAR(255) PRIMARY KEY,
    Name VARCHAR(255),
    Email VARCHAR(255),
    WebSite VARCHAR(255),
    Description VARCHAR(255),
    Cookie VARCHAR(1255),
    Provider VARCHAR(255)
);

CREATE TABLE IF NOT EXISTS DON_SERVICE (
    Id SERIAL PRIMARY KEY,
    ServiceId VARCHAR(255),
    Name VARCHAR(255),
    Field VARCHAR(255),
    Url VARCHAR(255),
    Description VARCHAR(255),
    Profile VARCHAR(255)
);
